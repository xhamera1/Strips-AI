(define (problem 60-states-blocksworld-rebuild)
  (:domain classic-blocksworld)
  
  (:init 
    (block a) (block b) (block c)
    (table t1) (table t2) (table t3)
    
    ;; Wieża początkowa na t1 (A na samym dole, C na szczycie)
    (on a t1) 
    (on b a) 
    (on c b)
    
    ;; Definiujemy, co jest aktualnie wolne od góry
    (clear c)   ;; Szczyt wieży
    (clear t2)  ;; Puste miejsce tymczasowe
    (clear t3)  ;; Puste miejsce docelowe
  )
  
  (:goal 
    ;; Chcemy identycznej wieży, ale na stole t3
    (and 
      (on a t3) 
      (on b a) 
      (on c b)
    )
  )
)
